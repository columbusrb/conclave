def oauth_settings_for(provider)
  key    = find_by_setting("#{provider}_oauth_key").try(:value)
  secret = find_by_setting("#{provider}_oauth_secret").try(:value)
  {key: key, secret: secret} if key && secret
end

def paperclip_settings
